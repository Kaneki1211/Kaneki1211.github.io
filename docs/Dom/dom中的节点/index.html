二、dom 中的节点

1、对象属性的获取​

var obj = {
  myname: "李白",
};
var keyname = "myname";
// 对象[属性]可以传变量  而通过对象.属性不识别变量
console.log(obj.myname); // 李白
console.log(obj["myname"]); //李白
console.log(obj[keyname]); //李白
console.log(obj["keyname"]); //undefined
console.log(obj[myname]); //undefined
console.log(obj.keyname); //undefined

2、元素的属性操作​

属性：attributes​

<div class="mydiv" id="myid" style="width: 100px" myattr="hello">我是div</div>
<button class="btn1"></button>
<button class="btn2"></button>
<button class="btn3"></button>
<ul class="container"></ul>

获取属性​

元素.getAttribute("属性名")

const btn1 = document.querySelector(".btn1");
const mydiv = document.querySelector(".mydiv");
btn1.onclick = function () {
  var myattr = mydiv.getAttribute("myattr");
  console.log(myattr); //hello
};

设置属性​

元素.setAttribute("属性名","属性值")

const btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  mydiv.setAttribute("mykey", "key");
};

移除一个元素的属性​

元素.removeAttribute("属性名")

const btn3 = document.querySelector(".btn3");
btn3.onclick = function () {
  mydiv.removeAttribute("style");
};

3、节点​

页面通过节点构成
节点（node）：页面上所有的内容都是节点
节点包括 注释 文本 元素（标签） 属性 ...
页面由节点构成，元素是节点的一类，称为元素节点

节点  第一：节点的类型 通过 nodeType来查看
      第二：节点名称 通过 nodeName来获取
      第三：节点的值 通过nodeValue来获取
<div class="container" class="box" id="wrap" my="mydiv">
  我是文本
  <span>我是span</span>
  <!-- 我是注释 -->
</div>

获取一个元素的字节点​

元素.childNodes

var container = document.querySelector(".container");
var nodes = container.childNodes;
console.log(nodes);
var textNode = nodes[0];
console.log(textNode);
console.log(typeof textNode);

文本节点​

//   文本节点的类型是 nodeType===3
console.log(textNode.nodeType);
//   文本节点的节点名称：nodeName：#text
console.log(textNode.nodeName);
//   文本节点的节点值：nodeValue：就是文本节点的内容 是字符串
console.log(textNode.nodeValue);
// 注意：空格换行也属于文本节点

元素节点​

var ele = node[1];
//   元素节点的类型：nodeType===1
console.log(nodeType);
//   元素的节点名：nodeName=="SPAN"
//   注意：元素的节点名称是标签名称 且全部大写
console.log(ele.nodeName);
//   元素节点的节点值：nodeValue===null
console.log(ele.nodeValue);

注释节点​

var commentNode = nodes[3];
//   节点类型：nodeType===8
console.log(commentNode.nodeType);
//   节点名称：nodeName===#comment
console.log(commentNode.nodeName);
//   节点值：nodeValue===注释的内容
console.log(commentNode.nodeValue);

属性节点​

// 获取属性节点
// 先获取元素 然后再通过元素获取元素的属性节点
console.log(div);
//   通过元素获取属性节点的伪数组：元素.attributes
var attrs = div.attributes;
console.log(attrs);
var myattr = attrs[1];
//   节点类型：nodeType===2
console.log(myattr.nodeType);
//   节点名称：nodeName===属性名
console.log(myattr.nodeName);
//   节点值：nodeValue===属性值
console.log(myattr.nodeValue);
