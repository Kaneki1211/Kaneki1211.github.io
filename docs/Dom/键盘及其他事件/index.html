六、键盘及其他事件

1、键盘事件​

  • keydown:键盘按下的时候执行
  • keypress:键盘按下的时候执行
  • keyup:键盘抬起的时候执行
  • 当你按下键盘的时候 会经历 keydown -keypress-keyup

键盘按下事件 keydown​

document.onkeydown = function () {
  console.log("键盘按下了");
};

键盘按下事件 keypress​

document.onkeypress = function () {
  console.log("keypress键盘按下了");
};

按键抬起事件 keyup​

document.onkeyup = function () {
  console.log("keypress键盘抬起了");
};

区分每个按键​

  • 第一种 通过事件对象的 keyCode（不推荐） 注意：keydown 和 keyup 按键字母不区分大小写
document.onkeydown = function (e) {
  var keyCode = e.keyCode;
  console.log(keyCode);
};
// keypress 会区分大小写
document.onkeypress = function (e) {
  var keyCode = e.keyCode;
  console.log(keyCode);
};
  • 第二种 通过 key 值获取按键
document.onkeydown = function (e) {
  var key = e.key;
  console.log(key);
};

组合按键​

// ctrl +按键：shift+按键；alt+按键

document.onkeydown = function (e) {
  // 按下alt和a的时候触发
  var key = e.key;
  // if (key == "a" && e.altKey) {
  //   console.log("按了alt+a");
  // }
  // 和系统按键冲突
  if (key == a && e.shiftKey) {
    console.log(111);
  }
};

2、表单事件​

表单输入事件 input​

var myinputEle = document.querySelector(".myinput");
myinputEle.oninput = function () {
  console.log("输入的时候执行");
};

聚焦事件 focus​

myinputEle.onfocus = function () {
  console.log("聚焦了");
};

失焦事件 blur​

//   单纯的失去焦点
myinputEle.onblur = function () {
  console.log("失去焦点");
};

表单改变事件 change​

// 第一：内容需要改变   第二：失去焦点
myinputEle.onchange = function () {
  console.log("改变了");
};

3、移动端事件​

开始触摸事件 touchstart​

var boxEle = document.querySelector(".box");
boxEle.ontouchstart = function () {

};

触摸移动事件 touchmove​

boxEle.ontouchmove = function () {
  console.log("移动的时候触发");
};

结束触摸事件 touchend​

boxEle.ontouchend = function () {
  console.log("结束触摸的时候触发");
};

4、其他事件​

过渡结束事件 transitionend​

  }, 2000);

};

5、零级事件和二级事件​

1. 0 级事件：onclick onmouseover ...
2. 1 级事件：没有
3. 2 级事件：addEventListener：添加事
4. 3 级事件：blur focus load resize ...

零级事件的绑定和解除​

// 0级事件：绑定方式：事件源.on+事件名称=事件函数
var btn1 = document.querySelector(".btn1");
btn1.onclick = function () {
  console.log("执行了");
};
// 解除绑定0级事件
var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  btn1.onclick = null;
};

二级事件的绑定和解除​

var btn3 = document.querySelector(".btn3");
var fn = () => {
  console.log("点击了");
};
btn3.addEventListener("click", fn);
//   解除绑定
// 解除绑定的时候的事件函数 必须和绑定时候的事件函数一样
var btn4 = document.querySelector(".btn4");
btn4.addEventListener("click", () => {
  btn3.removeEventListener("click", fn);
});

二级事件的兼容性​

// addEventListener符合IE8以上浏览器

//   主流浏览器
var boxEle = document.querySelector(".box");
boxEle.addEventListener("click", () => {
  console.log("点击了");
});
// IE8及以下
boxEle.addEventListener("onclick", () => {
  console.log("点击了");
});
// 兼容性的函数
//   eventName 事件名称：fn是事件函数 ；ele是事件源
function addEvent(ele, eventName, fn) {
  if (window.addEventListener) {
    // 有addEventListener，浏览器是主流浏览器
    ele.addEventListener(eventName, fn);
  } else {
    // 没有addEventListener  ，ie8及一下
    ele.attachEvent("on" + eventName, fn);
  }
}

addEvent(boxEle, "click", function () {
  console.log(123);
});

零级事件和二级事件的区别​

  • 2 级事件可以绑定多个事件 事件函数不会被覆盖 都会被执行
  • 0 级事件绑定多个事件的时候后一个会覆盖前面的事件函数
boxEle.addEventListener("click", () => {
  console.log(1111);
});
boxEle.addEventListener("click", () => {
  console.log(2222);
});

6、自定义事件​

// 自定义事件的绑定：事件名称自定义
var btn = document.querySelector("button");
// 事件名称都是预定义的，系统已经定义好的事件名称

// 绑定自定义事件需要借助一个内置类
var eventObj = new EventTarget();
// 添加自定义事件
// 事件名称可以自定义
eventObj.addEventListener("myevent", function () {
  console.log("自定义事件myevent执行了");
});

// 触发自定义事件
setTimeout(function () {
  eventObj.dispatchEvent(new CustomElement("myevent"));
}, 3000);

7、事件流​

先经历捕获阶段 ---->事件触发 --->事件冒泡​

0 级事件的执行​

// 0级事件事件流  默认事件流是 冒泡执行事件 ；
window.onclick = function () {
  console.log("window");
};

document.onclick = function () {
  console.log("document");
};

document.documentElement.onclick = function () {
  console.log("html");
};

document.body.onclick = function () {
  console.log("body");
};

2 级事件的执行​

// 二级事件的执行
// 二级事件默认也是冒泡阶段执行事件
// 可以修改 事件的执行过程 ，通过 事件绑定的第三个参数，默认是false 冒泡阶段执行事件
// 如果想在捕获阶段执行事件 把第三个参数设置成 true

// 1. 获取事件触发的元素 e.target  2. 阻止冒泡(捕获)  方式一：e.stopPropagation(); 方式二 ： e.cancelBubble = true;  3. 查看事件触发的路径 e.path
window.addEventListener("click", function (e) {
  console.log(e.target);
  // console.log("window");
});
document.addEventListener("click", function () {
  console.log("document");
});
document.documentElement.addEventListener("click", function () {
  console.log("html");
});
document.body.addEventListener("click", function () {
  console.log("body");
});
document.querySelector(".outer").addEventListener("click", function (e) {
  console.log("outer");
  // e.stopPropagation();
});
document.querySelector(".center").addEventListener("click", function (e) {
  console.log("center");
  // e.stopPropagation();  // 阻止事件的冒泡
  // e.cancelBubble = true;
  console.log(e.path);
});
