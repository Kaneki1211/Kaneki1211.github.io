一、dom 的操作

1、dom 的获取​

<div id="mydiv">我是div<span>我是div里的span</span></div>
<span class="myspan">我是span标签</span>
<p>我是p标签111</p>
<p>我是p标签222</p>
<ul>
  <li></li>
  <li></li>
  <li></li>
  <li></li>
</ul>
  • 通过 id 名获取元素节点
var myDivEle = document.getElementById("mydiv");
  • 通过 class 名获取元素节点
    • 会获得一个伪数组 HTMLCollection 不能使用数组的 Api
    • 通过 Elements 获取的是一个伪数组，会把元素放在伪数组里
var mySpanEle = document.getElementsByClassName("myspan");
  • 通过标签名称获取元素
var pEles = document.getElementsByTagName("p");

  • 通过 css 选择器获取元素

    • 获取单个元素 querySelector()
    /* 通过类名获取单个元素节点 */
    var ele = document.querySelector(".myspan");
    /* 通过id名获取单个元素节点 */
    var div = document.querySelector(".#mydiv");
    // document 是范围，表示在document范围内获取元素
    // 缩小范围
    var span = div.querySelector("span");
    /* 注意：即使有多个元素符合条件，nm通过querySelector也只能获取第一个元素 */
    • 获取多个元素 querySelectorAll()
    var ele = document.querySelector("span");
    console.dir(ele); //深层次打印数据
    /*
     通过querySelectorAll()获取的也是一个伪数组
     即使只有一个元素也会返回一个伪数组
    */

2、伪数组​

  • 伪数组和数组类似但不能使用数组的 api
  • 将伪数组转换为数组
    • var 数组=[...伪数组]
    • var 数组=Array.form(伪数组)

3、操作元素里的内容​

  • element.textContent 获取或修改元素中的文本内容
    • 获取的是标签中的内容，不会考虑 css 样式
    • 当字符串中有标签时，会自动对标签进行转义
  • element.innerText 获取或修改元素中的文本内容
    • innerText 获取内容时，会考虑 css 样式
    • 当字符串中有标签时，会自动对标签进行转义
    • 通过 innerText 去读取 CSS 样式，会触发网页的重排（计算 CSS 样式）
  • element.innerHTML 获取或修改元素中的 html 代码
    • 可以直接向元素中添加 html 代码
    • innerHTML 插入内容时，有被 xss 注入的风险

4、行内样式的操作​

获取行内样式​

注意 ：获取元素样式的时候 ，只能获取行间样式 不能获取 内嵌 和外链的样式 ；

<div class="box" style="color:red;width:300px;height:300px">我是boxdiv</div>
<button class="btn1">点击我获取样式</button>

<script>
var boxEle = document.querySelector(".box");
var btn1 = document.querySelector(".btn1");
btn1.onclick = function () {
  // 获取 boxEle的样式 ；
  // console.log(boxEle.style);
  var w = boxEle.style.width; // 获取 boxEle的宽度；有单位的；
  console.log(w);
};
</script>

设置行间样式​

注意： 设置样式的时候 ，如果是 带 - 的样式 js 设置的时候需要大写

<div class="box" style="color:red;width:300px;height:300px">我是boxdiv</div>
<button class="btn2">点击我获取样式</button>

<script>
var btn2 = document.querySelector(".btn2");
var boxEle = document.querySelector(".box");
btn2.onclick = function () {
  // 设置 类名为 box的 div 的 字体的颜色 ；
  boxEle.style.color = "white";
  // boxEle.style.background-color = "blue"; // 错误的写法
  boxEle.style.backgroundColor = "blue";
  // background-color  ------> js ： backgroundColor 去掉- 然后把-后面的字符大写
};
</script>

5、类名操作​

通过 className 操作​

获取元素的类名​

<div class="box2 active">我是div</div>
<button class="btn3">点击我获取类名</button>


var btn3 = document.querySelector(".btn3");
var box2 = document.querySelector(".box2");
btn3.onclick = function () {
  // 获取 box2元素的类名
  var className = box2.className;
  // 可以获取 元素的所有类名，得到的结果是一个字符串；
  console.log(className);
};

设置元素的类名​

  注意 ： 设置类名会覆盖之前的类名。会导致 之前的样式丢失
<div class="box2 active">我是div</div>
<button class="btn4">点击我设置类名</button>

var btn4  = document.querySelector(".btn4");
var box2 = document.querySelector(".box2");
btn4.onclick = function(){
// box2.className = "show";  // 设置 box2 元素的类名；
// 在之前的类名基础上添加 类名

box2.className += " show"; // 注意拼接的时候 需要空格；
}

通过类名对象操作​

js 里每个元素都会提供一个操作类名的对象，这个对象叫 classList，提供各种方法来操作类名

<div class="box cative">我是div</div>
<button class="btn1">点击1</button>
<button class="btn2">点击添加类名</button>
<button class="btn3">点击移除类名</button>
<button class="btn4">点击检查类名</button>
<button class="btn5">点击添加或删除类名</button>

1.获取元素类名的个数:元素.classList.length​

var boxEle = document.querySelector(".box");
var btn1 = document.querySelector(".btn1");

btn1.onclick = function () {
  console.log(boxEle.classList.length);
};

2.给元素添加类名： 元素.classList.add("类名")​

  • className 赋值修改类名是一个覆盖操作，会把之前的类名覆盖掉
  • classList 提供的添加方法是类名的追加操作
var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  boxEle.classList.add("bg");
};

3.将元素的类名删除：元素.classList.remove("类名")​

var btn3 = document.querySelector(".btn3");
btn3.onclick = function () {
  boxEle.classList.remove("bg");
};

4.判断元素是否包含某个类名： 元素.classList.contains("类名")​

var btn4 = document.querySelector(".btn4");
btn4.onclick = function () {
  var flag = boxEle.classList.contains("bg");
  flag ? console.log("包含") : console.log("不包含");
};

5.判断元素是否有某个类名，如果有就删除，否则就添加这个类名: 元素.classList.toggle("类名")​

var btn5 = document.querySelector(".btn5");
btn5.onclick = function () {
  boxEle.classList.toggle("mytest");
};

6、JS 中几个特殊的标签​

html 标签 简写形式​

var htmlEle = document.documentElement; //html标签
console.log(htmlEle);

body 标签 简写形式​

var bodyEle = document.body;
console.log(bodyEle);

获取 head 标签 简写形式​

var headEle = document.head;
console.log(headEle);

获取 title 标签里的内容 简写形式​

注意：document.title不是获取title标签 而是获取或设置title标签里的内容（innerText）
var titlestr = document.title;
console.log(titlestr);
// 设置title内容
document.title = "hello";
