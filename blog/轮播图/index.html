function Dots() {
  var lis = document.querySelectorAll("li");

轮播图功能​

autoPlay();
function autoPlay() {
  timer = setInterval(function () {
    // -520 * (index + 1) - 520  第一次图片由left：-520px 到 left：-1040px 移动
    move(imgContainerEle, "left", -520 * (index + 1) - 520, function () {
      index++;
      if (index == 5) {
        index = 0;
        imgContainerEle.style.left = "-520px";
      }
      //   每次图片改变都需要判断
      Dots();
    });
  }, 2000);
}

鼠标移入停止轮播图 移出开启轮播图​

// 鼠标移入的时候 停止轮播的定时器
containerEle.onmouseenter = function () {
  console.log("执行了");
  clearInterval(timer);
};

// 鼠标移除的时候 轮播定时器继续开启；
containerEle.onmouseleave = function () {
  autoPlay();
};

// 点击下一页按钮 轮播图切换到下一页；
var nextEle = document.querySelector(".next");
// debounce 防抖 一定时间内连续点击多次只会生效一次
nextEle.onclick = debounce(function () {
  move(imgContainerEle, "left", -520 * (index + 1) - 520, function () {
    index++;
    console.log(index);
    if (index == 5) {
      index = 0;
      imgContainerEle.style.left = "-520px";
    }
    Dots();
  });
}, 1000);

      // 移动函数
      function move(boxEle, dir, target, cb) {
        var timer4 = setInterval(function () {
          var num = parseInt(getComputedStyle(boxEle)[dir]);
          var n = target > num ? 1 : -1;
          num += 5 * n;
          boxEle.style[dir] = num + "px";
          if (num == target) {
            clearInterval(timer4);
            console.log("结束");
            cb();
          }
        }, 10);
      }

      //防抖函数
      function debounce(fn, delay) {
        var timer;
        return function () {
          if (!timer) {
            fn();
          }
          clearTimeout(timer);
          timer = setTimeout(() => {
            timer = null;
          }, delay);
        };
      }
