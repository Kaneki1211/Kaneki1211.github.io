DOM 方块拖拽

2023年1月10日 · 阅读需 9 分钟

单方块拖拽​

拖动后事件源的位置

x 轴的位置  B.clientX - A.clientX + A.offsetLeft
y 轴的位置  B.clientY - A.clientY + A.offsetTop

  • 鼠标按下的时候开始拖动 :onmousedown

  • 鼠标移动的时候元素开始动 ； onmousemove

  • 鼠标抬起的时候 停止拖动 onmouseup

<style>
  .box1 {
    width: 200px;
    height: 200px;
    background: red;
    position: absolute;
  }
</style>

<div class="box1"></div>
var div = document.querySelector(".box1");
div.onmousedown = function (e) {
  var x = e.clientX; // 绿色
  var y = e.clientY;
  var left = this.offsetLeft; // 紫色
  var top = this.offsetTop;
  div.onmousemove = function (e) {
    div.style.left = e.clientX - x + left + "px"; //把红线的位置 设置成box的坐标
    div.style.top = e.clientY - y + top + "px";
  };
};
div.onmouseup = function () {
  div.onmousemove = ""; //给onmousemove赋值为空 覆盖原有的值
};

divs.forEach((item) => {

    var x = e.clientX - this.offsetLeft;

      this.style.left = e.clientX - x + "px";
      this.style.top = e.clientY - y + "px";
    };
  };
  item.onmouseup = function () {
    this.onmousemove = "";
    this.style.zIndex = 0;
  };
});

检测两个方块是否碰撞​

碰撞条件：
    A的最大边Amax 大于 B的最小边Bmin
    A的最小边Amin 小于 B的最大边Bmax
    两个条件同时满足

<style>
  .box1 {
    width: 100px;
    height: 100px;
    background-color: red;
    position: absolute;
    left: 0px;
  }

  .box2 {
    width: 100px;
    height: 100px;
    background-color: blue;
    position: absolute;
    left: 300px;
  }
</style>

<div class="box1"></div>
<div class="box2"></div>
var box1Ele = document.querySelector(".box1");
var box2Ele = document.querySelector(".box2");

box1Ele.onmousedown = function (e) {
  var x = e.clientX - this.offsetLeft;
  var y = e.clientY - this.offsetTop;
  this.onmousemove = function (e) {
    // 在方块移动的时候需要检测是否碰了 ；
    if (isCrash(box1Ele, box2Ele)) {
      console.log("碰撞了");
    } else {
      console.log("没有碰撞");
    }
    this.style.left = e.clientX - x + "px";
    this.style.top = e.clientY - y + "px";
  };
};
box1Ele.onmouseup = function () {
  this.onmousemove = "";
};
//  把碰撞封装成函数  提高复用性
function isCrash(box1Ele, box2Ele) {
  var Almin = box1Ele.offsetLeft;
  var Almax = box1Ele.offsetLeft + box1Ele.offsetWidth;
  var Blmin = box2Ele.offsetLeft;
  var Blmax = box2Ele.offsetLeft + box2Ele.offsetWidth;
  var Atmin = box1Ele.offsetTop;
  var Atmax = box1Ele.offsetTop + box1Ele.offsetHeight;
  var Btmin = box2Ele.offsetTop;
  var Btmax = box2Ele.offsetTop + box2Ele.offsetHeight;
  if (Almax > Blmin && Almin < Blmax && Atmax > Btmin && Atmin < Btmax) {
    return true;
  } else {
    return false;
  }
}

多方块碰撞 碰撞面积最大交换位置​

计算碰撞面积​

检查拖拽元素和其他 8 个元素是否碰撞

// 1.检测拖拽元素 和其他8个元素是否碰撞了
var that = this;
divs.forEach(function (div) {
  if (div !== that) {
    // 把自己排除；
    // 其他的8个元素；
    // 检测this 和其他的8个元素是否碰了；
    if (isCrash(that, div)) {
      // console.log("有div和this碰了",div);
      crashArr.push(div);
    }
  }
});

获取碰撞最大面积的元素

var maxArea = 0;
var maxEle = null;
crashArr.forEach((item) => {
  var w = this.offsetWidth - Math.abs(item.offsetLeft - this.offsetLeft);
  var h = this.offsetHeight - Math.abs(item.offsetTop - this.offsetTop);
  var area = w * h;
  if (area > maxArea) {
    maxArea = area;
    maxEle = item;
    console.log(maxArea);
  }
});

和碰撞面积最大的交换位置​

var position = {};
position.left = this.style.left;
position.top = this.style.top;
/* 鼠标抬起时判断 */
item.onmouseup = function () {
  this.style.transition = "1s";
  maxEle.style.transition = "1s";
  if (maxEle) {
    // 和碰撞元素交换位置
    this.style.top = maxEle.style.top;
    this.style.left = maxEle.style.left;
    maxEle.style.top = position.top;
    maxEle.style.left = position.left;
  } else {
    // 拖拽元素没有碰撞元素；
    this.style.left = initPos.left;
    this.style.top = initPos.top;
  }
  this.onmousemove = "";

  this.style.zIndex = 0;
};

for (var i = 0; i < divs.length; i++) {

var containerEle = document.querySelector(".container");
// console.log( divs);
divs.forEach(function (item) {
  // item 是 每一个 方块div
  item.onmousedown = function (e) {
    // 开始拖动的时候 把 拖拽元素的过渡效果取消；
    this.style.transition = "none";
    // 当鼠标按下的时候 把这个元素的层级提高
    this.style.zIndex = 10;
    var x = e.clientX - this.offsetLeft;
    var y = e.clientY - this.offsetTop;
    var initPos = {}; // 记录拖拽元素的初始位置；
    initPos.left = this.style.left;
    initPos.top = this.style.top;
    console.log(initPos);
    var maxEle = null; // 碰撞面积最大的元素
    this.onmousemove = function (e) {
      var crashArr = []; // 存放和拖拽元素碰撞了的元素；
      // 检测拖动元素 和 其他8个元素是否碰撞了，
      // 如果碰了 ，就把碰撞的元素保存起来
      //  判断 碰撞元素的面积 ，让拖拽元素和碰撞面积最大的元素互换位置
      // 1.检测拖拽元素 和其他8个元素是否碰撞了
      var that = this;
      divs.forEach(function (div) {
        // console.log("?",div);
        if (div !== that) {
          // 把自己排除；
          // console.log(123);
          // 其他的8个元素；
          // 检测this 和其他的8个元素是否碰了；
          if (isCrash(that, div)) {
            // console.log("有div和this碰了",div);
            crashArr.push(div);
          }
        }
      });
      console.log("碰撞的元素数组", crashArr);
      // 计算和碰撞元素的面积
      var maxArae = 0; // 假定最大碰撞面积是0；
      crashArr.forEach(function (item) {
        var w = that.offsetWidth - Math.abs(item.offsetLeft - that.offsetLeft);
        var h = that.offsetHeight - Math.abs(item.offsetTop - that.offsetTop);
        var arae = w * h;
        if (arae > maxArae) {
          maxArae = arae;
          // 获取最大面积
          // console.log("最大面积碰撞的")
          maxEle = item;
        }
      });
      console.log("碰撞最大面积的元素是:", maxEle);
      var disX = e.clientX - x;
      var disY = e.clientY - y;
      // 左边的边距
      if (disX <= 0) {
        disX = 0;
      }
      // 右边的边距
      if (disX >= containerEle.clientWidth - this.offsetWidth) {
        disX = containerEle.clientWidth - this.offsetWidth;
      }
      // 上边的位置
      if (disY <= 0) {
        disY = 0;
      }
      // 下边的位置
      if (disY >= containerEle.clientHeight - this.offsetHeight) {
        disY = containerEle.clientHeight - this.offsetHeight;
      }
      this.style.left = disX + "px";
      this.style.top = disY + "px";
    };
    this.onmouseup = function () {
      // 互换位置 ： 拖拽元素  和 碰撞面积最大的元素   位置；
      // 给拖拽的元素加上过渡；
      this.style.transition = "all 1s";
      if (maxEle) {
        // 有碰撞最大面积元素
        this.style.left = maxEle.style.left;
        this.style.top = maxEle.style.top;

        maxEle.style.left = initPos.left;
        maxEle.style.top = initPos.top;
      } else {
        // 拖拽元素没有碰撞元素；
        this.style.left = initPos.left;
        this.style.top = initPos.top;
      }
      this.style.zIndex = 0;
      this.onmousemove = "";
    };
  };
});
