单方块拖拽​

拖动后事件源的位置

x 轴的位置  B.clientX - A.clientX + A.offsetLeft
y 轴的位置  B.clientY - A.clientY + A.offsetTop

  • 鼠标按下的时候开始拖动 :onmousedown

  • 鼠标移动的时候元素开始动 ； onmousemove

  • 鼠标抬起的时候 停止拖动 onmouseup

<style>
  .box1 {
    width: 200px;
    height: 200px;
    background: red;
    position: absolute;
  }
</style>

<div class="box1"></div>
var div = document.querySelector(".box1");
div.onmousedown = function (e) {
  var x = e.clientX; // 绿色
  var y = e.clientY;
  var left = this.offsetLeft; // 紫色
  var top = this.offsetTop;
  div.onmousemove = function (e) {
    div.style.left = e.clientX - x + left + "px"; //把红线的位置 设置成box的坐标
    div.style.top = e.clientY - y + top + "px";
  };
};
div.onmouseup = function () {
  div.onmousemove = ""; //给onmousemove赋值为空 覆盖原有的值
};

检测两个方块是否碰撞​

碰撞条件：
    A的最大边Amax 大于 B的最小边Bmin
    A的最小边Amin 小于 B的最大边Bmax
    两个条件同时满足

var box1Ele = document.querySelector(".box1");
var box2Ele = document.querySelector(".box2");

box1Ele.onmousedown = function (e) {
  var x = e.clientX - this.offsetLeft;
  var y = e.clientY - this.offsetTop;
  this.onmousemove = function (e) {
    // 在方块移动的时候需要检测是否碰了 ；
    if (isCrash(box1Ele, box2Ele)) {
      console.log("碰撞了");
    } else {
      console.log("没有碰撞");
    }
    this.style.left = e.clientX - x + "px";
    this.style.top = e.clientY - y + "px";
  };
};
box1Ele.onmouseup = function () {
  this.onmousemove = "";
};
//  把碰撞封装成函数  提高复用性
function isCrash(box1Ele, box2Ele) {
  var Almin = box1Ele.offsetLeft;
  var Almax = box1Ele.offsetLeft + box1Ele.offsetWidth;
  var Blmin = box2Ele.offsetLeft;
  var Blmax = box2Ele.offsetLeft + box2Ele.offsetWidth;
  var Atmin = box1Ele.offsetTop;
  var Atmax = box1Ele.offsetTop + box1Ele.offsetHeight;
  var Btmin = box2Ele.offsetTop;
  var Btmax = box2Ele.offsetTop + box2Ele.offsetHeight;
  if (Almax > Blmin && Almin < Blmax && Atmax > Btmin && Atmin < Btmax) {
    return true;
  } else {
    return false;
  }
}
